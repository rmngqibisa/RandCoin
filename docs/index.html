def mine_pending_transactions(self, miner_address):
    """Mines transactions and rewards the miner."""

    block = Block(
        len(self.chain),
        self.get_latest_block().hash,
        self.timestamp,
        self.pending_transactions,
        self.difficulty
    )

    block.mine() # Proof of Work
    self.chain.append(block)
